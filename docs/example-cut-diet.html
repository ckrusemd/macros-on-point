2.3 Clean data table

df_clean = df %>% 
  slice(2:nrow(.)) %>% 
  dplyr::select(-X1) %>% 
  dplyr::rename(Group=X2,
                Name=X3)
df_clean %>% 
  head() %>% 
  DT::datatable(options = list(scrollX = TRUE))

2.4 Tidy Data

df_basic_tidy = df_clean %>% 
  dplyr::select(Group,
                Name,
                `Energy,.kcal`,
                `Protein,.videnskabelig`,
                `Carbohydrate,.declaration`,
                `Dietary.fiber`,
                `Fat,.total`,
                Alcohol) %>% 
  setNames(.,c("Group","Name","Kcal","Protein","Carbohydrates","Fiber","Fat","Alcohol")) %>% 
  group_by(Group,Name) %>% 
  filter(n()==1) %>% 
  ungroup() %>% 
  gather(Metric,Value,Kcal:Alcohol) %>% 
  dplyr::mutate(Value=as.numeric(Value)) 
## Warning in mask$eval_all_mutate(quo): NAs introduced by coercion
df_basic_tidy %>% 
  head() %>% 
  DT::datatable(options = list(scrollX = TRUE))

2.5 Spread data

df_basic = df_basic_tidy %>% 
  spread(Metric,Value)

df_basic %>% 
  head() %>% 
  DT::datatable(options = list(scrollX = TRUE))
